6.1 - A Vector Type

                
                    class Vec {
                        constructor(x, y) {
                            this.x = x;
                            this.y = y;
                        }
                        get length() {
                            return Math.sqrt((this.x * this.x) + (this.y * this.y));
                        }
                        plus(value) {
                            this.x += value.x;
                            this.y += value.y;
                            return this;
                        }
                        minus(value) {
                            this.x -= value.x;
                            this.y -= value.y;
                            return this;
                        }
                    };
    
                    console.log(new Vec(1, 2).plus(new Vec(2, 3)));
                    console.log(new Vec(1, 2).minus(new Vec(2, 3)));
                    console.log(new Vec(3, 4).length);
