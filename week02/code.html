2.2 - FizzBuzz

                        let num;
                        if (i % 5 === 0 && i % 3 === 0) num = "FizzBuzz";
                        else if (i % 3 === 0) num = "Fizz";
                        else if (i % 5 === 0) num = "Buzz";
                        else num = i;
                        console.log(num);
                    }
